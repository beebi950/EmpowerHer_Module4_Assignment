CREATE TABLE students (
  id SERIAL PRIMARY KEY,
  name TEXT NOT NULL,
  email TEXT UNIQUE NOT NULL,
  age INTEGER,
  course TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);
